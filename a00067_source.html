19 // Part of the implementation of the Super 4-points Congruent Sets (Super 4PCS)
20 // algorithm presented in:
21 //
22 // Super 4PCS: Fast Global Pointcloud Registration via Smart Indexing
23 // Nicolas Mellado, Dror Aiger, Niloy J. Mitra
24 // Symposium on Geometry Processing 2014.
25 //
26 // Data acquisition in large-scale scenes regularly involves accumulating
27 // information across multiple scans. A common approach is to locally align scan
28 // pairs using Iterative Closest Point (ICP) algorithm (or its variants), but
29 // requires static scenes and small motion between scan pairs. This prevents
30 // accumulating data across multiple scan sessions and/or different acquisition
31 // modalities (e.g., stereo, depth scans). Alternatively, one can use a global
32 // registration algorithm allowing scans to be in arbitrary initial poses. The
33 // state-of-the-art global registration algorithm, 4PCS, however has a quadratic
34 // time complexity in the number of data points. This vastly limits its
35 // applicability to acquisition of large environments. We present Super 4PCS for
36 // global pointcloud registration that is optimal, i.e., runs in linear time (in
37 // the number of data points) and is also output sensitive in the complexity of
38 // the alignment problem based on the (unknown) overlap across scan pairs.
39 // Technically, we map the algorithm as an ‘instance problem’ and solve it
40 // efficiently using a smart indexing data organization. The algorithm is
41 // simple, memory-efficient, and fast. We demonstrate that Super 4PCS results in
42 // significant speedup over alternative approaches and allows unstructured
43 // efficient acquisition of scenes at scales previously not possible. Complete
44 // source code and datasets are available for research use at
45 // http://geometry.cs.ucl.ac.uk/projects/2014/super4PCS/.

303 /*!
304 
305  This algorithm uses the simple distance to the split plane to prune nodes.
306  A more elaborated approach consists to track the closest corner of the cell
307  relatively to the current query point. This strategy allows to save about 5%
308  of the leaves. However, in practice the slight overhead due to this tracking
309  reduces the overall performance.
310 
311  This algorithm also use a simple stack while a priority queue using the squared
312  distances to the cells as a priority values allows to save about 10% of the leaves.
313  But, again, priority queue insertions and deletions are quite involved, and therefore
314  a simple stack is by far much faster.
315 
316  The optionnal parameter currentId is used when the query point is
317  stored in the tree, and must thus be avoided during the query
318 */
319 template<typename Scalar, typename Index>
320 template<int stackSize>
321 std::pair<Index, Scalar>
322 KdTree<Scalar, Index>::doQueryRestrictedClosestIndex(
323  RangeQuery<stackSize>& query,
324  int currentId) const
325 {
326 
327  Index cl_id = invalidIndex();
328  Scalar cl_dist = query.sqdist;
329 
330  query.nodeStack[0].nodeId = 0;
331  query.nodeStack[0].sq = 0.f;
332  unsigned int count = 1;
333 
334  //int nbLoop = 0;
335  while (count)
336  {
337  //nbLoop++;
338  QueryNode& qnode = query.nodeStack[count-1];
339  const KdNode& node = mNodes[qnode.nodeId];
340 
341  if (qnode.sq < cl_dist)
342  {
343  if (node.leaf)
344  {
345  --count; // pop
346  const int end = node.start+node.size;
347  for (int i=node.start ; i<end ; ++i){
348  const Scalar sqdist = (query.queryPoint - mPoints[i]).squaredNorm();
349  if (sqdist <= cl_dist && mIndices[i] != currentId){
350  cl_dist = sqdist;
351  cl_id = mIndices[i];
352  }
353  }
354  }
355  else
356  {
357  // replace the stack top by the farthest and push the closest
358  const Scalar new_off = query.queryPoint[node.dim] - node.splitValue;
359 
360  //std::cout << "new_off = " << new_off << std::endl;
361 
362  if (new_off < 0.)
363  {
364  query.nodeStack[count].nodeId = node.firstChildId; // stack top the farthest
365  qnode.nodeId = node.firstChildId+1; // push the closest
366  }
367  else
368  {
369  query.nodeStack[count].nodeId = node.firstChildId+1;
370  qnode.nodeId = node.firstChildId;
371  }
372  query.nodeStack[count].sq = qnode.sq;
373  qnode.sq = new_off*new_off;
374  ++count;
375  }
376  }
377  else
378  {
379  // pop
380  --count;
381  }
382  }
383  return std::make_pair(cl_id, cl_dist);
384 }

407  const KdNode & node = mNodes[qnode.nodeId];
408 
409  if (qnode.sq < query.sqdist)

411  if (node.leaf)

414  unsigned int end = node.start+node.size;
415  for (unsigned int i=node.start ; i<end ; ++i)
416  if ( (query.queryPoint - mPoints[i]).squaredNorm() < query.sqdist){
417  f(i);
418  }
419  }
420  else

447 template<typename Scalar, typename Index>
448 unsigned int KdTree<Scalar, Index>::split(int start, int end, unsigned int dim, Scalar splitValue)
449 {
450  int l(start), r(end-1);
451  for ( ; l<r ; ++l, --r)
452  {
453  while (l < end && mPoints[l][dim] < splitValue)
454  l++;
455  while (r >= start && mPoints[r][dim] >= splitValue)
456  r--;
457  if (l > r)
458  break;
459  std::swap(mPoints[l],mPoints[r]);
460  std::swap(mIndices[l],mIndices[r]);
461  }
462  return (mPoints[l][dim] < splitValue ? l+1 : l);
463 }

465 /*!
466 
467  Recursively builds the kdtree
468 
469 
470  The heuristic is the following:
471  - if the number of points in the node is lower than targetCellsize then make a leaf
472  - else compute the AABB of the points of the node and split it at the middle of
473  the largest AABB dimension.
474 
475  This strategy might look not optimal because it does not explicitly prune empty space,
476  unlike more advanced SAH-like techniques used for RT. On the other hand it leads to a shorter tree,
477  faster to traverse and our experience shown that in the special case of kNN queries,
478  this strategy is indeed more efficient (and much faster to build). Moreover, for volume data
479  (e.g., fluid simulation) pruning the empty space is useless.
480 
481  Actually, storing at each node the exact AABB (we therefore have a binary BVH) allows
482  to prune only about 10% of the leaves, but the overhead of this pruning (ball/ABBB intersection)
483  is more expensive than the gain it provides and the memory consumption is x4 higher !
484 */
485 template<typename Scalar, typename Index>
486 void KdTree<Scalar, Index>::createTree(unsigned int nodeId, unsigned int start, unsigned int end, unsigned int level, unsigned int targetCellSize, unsigned int targetMaxDepth)
487 {
488 
489  KdNode& node = mNodes[nodeId];
490  AxisAlignedBoxType aabb;
491  //aabb.Set(mPoints[start]);
492  for (unsigned int i=start ; i<end ; ++i)
493  aabb.extend(mPoints[i]);
494 
495  VectorType diag = aabb.diagonal();
496  typename VectorType::Index dim;
497 
498 #ifdef DEBUG
499 
500 // std::cout << "createTree("
501 // << nodeId << ", "
502 // << start << ", "
503 // << end << ", "
504 // << level << ")"
505 // << std::endl;
506 
507  if (std::isnan(diag.maxCoeff(&dim))){
508  std::cerr << "NaN values discovered in the tree, abort" << std::endl;
509  return;
510  }
511 #else
512  diag.maxCoeff(&dim);
513 #endif
514 
515 
516 #undef DEBUG
517  node.dim = dim;
518  node.splitValue = aabb.center()(dim);
519 
520  unsigned int midId = split(start, end, dim, node.splitValue);
521 
522  node.firstChildId = mNodes.size();
523 
524  {
525  KdNode n;
526  n.size = 0;
527  mNodes.push_back(n);
528  mNodes.push_back(n);
529  }
530  //mNodes << Node() << Node();
531  //mNodes.resize(mNodes.size()+2);
532 
533  {
534  // left child
535  unsigned int childId = mNodes[nodeId].firstChildId;
536  KdNode& child = mNodes[childId];
537  if (midId-start <= targetCellSize || level>=targetMaxDepth)
538  {
539  child.leaf = 1;
540  child.start = start;
541  child.size = midId-start;
542  }
543  else
544  {
545  child.leaf = 0;
546  createTree(childId, start, midId, level+1, targetCellSize, targetMaxDepth);
547  }
548  }
549 
550  {
551  // right child
552  unsigned int childId = mNodes[nodeId].firstChildId+1;
553  KdNode& child = mNodes[childId];
554  if (end-midId <= targetCellSize || level>=targetMaxDepth)
555  {
556  child.leaf = 1;
557  child.start = midId;
558  child.size = end-midId;
559  }
560  else
561  {
562  child.leaf = 0;
563  createTree(childId, midId, end, level+1, targetCellSize, targetMaxDepth);
564  }
565  }
566 }
