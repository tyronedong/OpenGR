19 // An implementation of the Super 4-points Congruent Sets (Super 4PCS)
20 // algorithm presented in:
21 //
22 // Super 4PCS: Fast Global Pointcloud Registration via Smart Indexing
23 // Nicolas Mellado, Dror Aiger, Niloy J. Mitra
24 // Symposium on Geometry Processing 2014.
25 //
26 // Data acquisition in large-scale scenes regularly involves accumulating
27 // information across multiple scans. A common approach is to locally align scan
28 // pairs using Iterative Closest Point (ICP) algorithm (or its variants), but
29 // requires static scenes and small motion between scan pairs. This prevents
30 // accumulating data across multiple scan sessions and/or different acquisition
31 // modalities (e.g., stereo, depth scans). Alternatively, one can use a global
32 // registration algorithm allowing scans to be in arbitrary initial poses. The
33 // state-of-the-art global registration algorithm, 4PCS, however has a quadratic
34 // time complexity in the number of data points. This vastly limits its
35 // applicability to acquisition of large environments. We present Super 4PCS for
36 // global pointcloud registration that is optimal, i.e., runs in linear time (in
37 // the number of data points) and is also output sensitive in the complexity of
38 // the alignment problem based on the (unknown) overlap across scan pairs.
39 // Technically, we map the algorithm as an 'instance problem' and solve it
40 // efficiently using a smart indexing data organization. The algorithm is
41 // simple, memory-efficient, and fast. We demonstrate that Super 4PCS results in
42 // significant speedup over alternative approaches and allows unstructured
43 // efficient acquisition of scenes at scales previously not possible. Complete
44 // source code and datasets are available for research use at
45 // http://geometry.cs.ucl.ac.uk/projects/2014/super4PCS/.

126 /*!
127  * \brief Convert a normalized n-d vector to a linear index in a uniform regular grid
128  * This function is recursive, and unrolled at compile time (loop over n).
129  *
130  * \param coord Input coordinates defined in the normalized n-hypercube.
131  * \param cdim Working dimension, must be called with n.
132  * \param gsize Dimension of the grid, must be consistent in all dimensions
133  *
134  * \tparam validate Enable or disable the range validation
135  * \tparam PointT Type of the input points (deduced)
136  * \tparam IndexT Index type (deduced)
137  * \tparam IndexT Size type (deduced)
138  *
139  * \see internal::IndexValidator for the validation procedure
140  */
141 template<bool validate, class ndIndexT, class IndexT, class SizeT>
142 constexpr inline IndexT
143 UnrollIndexLoop(const ndIndexT& coord,
144  IndexT cdim,
145  SizeT gsize){
146  return (cdim != 0)
147  ? ( internal::IndexValidator<validate>::validate(IndexT(coord[cdim]), gsize)*POW(gsize, cdim) +
148  UnrollIndexLoop<validate>(coord, cdim-1, gsize) )
149  : internal::IndexValidator<validate>::validate(IndexT(coord[cdim]), gsize);
150 }

153 /*!
154  * \brief Convert a normalized n-d vector to a linear index in a uniform regular
155  * grid, moved by moved by an offset defined as a integer move in the n-d grid.
156  * This function is recursive, and unrolled at compile time (loop over n). In
157  * addition, it allows to offset the input coordinates.
158  *
159  * \param coord Input coordinates defined in the normalized n-hypercube.
160  * \param cdim Working dimension, must be called with n.
161  * \param gsize Dimension of the grid, must be consistent in all dimensions
162  *
163  * \see UnrollIndexLoop<class PointT>
164  */
165 template<bool validate, class ndIndexT, class IndexT, class SizeT>
166 constexpr inline IndexT
167 UnrollIndexLoop(const ndIndexT& coord,
168  const ndIndexT& offset,
169  IndexT cdim,
170  SizeT gsize){
171  return (cdim != 0)
172  ? ( internal::IndexValidator<validate>::validate(IndexT(offset[cdim]+std::floor(coord[cdim])), gsize)*POW(gsize, cdim) +
173  UnrollIndexLoop<validate>(coord, offset, cdim-1, gsize) )
174  : internal::IndexValidator<validate>::validate(IndexT(offset[cdim]+std::floor(coord[cdim])), gsize);
175 }

178 /// Compute the 3^dim neighborhood for a cell
179 /// \FIXME This implementation is not efficient and must be improved
180 /// e.g., do no allocate any array, just call a function with the ids.
181 struct OneRingNeighborhood {
182  /// helper class
183  template <int dim>
184  struct NeighborhoodType {
185  using type = std::array<int, Utils::POW(3, dim) >;
186  using ptr = int*;
187  };
188 
189  /// \param queryId Linear index of the query
190  /// \param nbElementPerDim Number of cells in the grid per dimension
191  /// Total size of the grid is `pow(nbElementPerDim, dim)`.
192  /// \return An array of the neighboring cells ids. When the query is
193  /// on the grid boundaries, out of bounds cells ids are marked
194  /// with -1 in the output array.
195  template <int dim>
196  void get(
197  int queryId,
198  int nElPerDim,
199  typename NeighborhoodType<dim>::type& nei) {
200  get<dim> ( queryId, nElPerDim, 0, nei.data(), nei.data()+nei.size() );
201  }
202 
203 private:
204  template <int dim>
205  void get(
206  int /*queryId*/,
207  int /*nElPerDim*/,
208  int /*offset*/,
209  typename NeighborhoodType<dim>::ptr first,
210  typename NeighborhoodType<dim>::ptr last) {
211  std::fill(first, last, -1);
212  }
213 };

216 template<>
217 void
218 OneRingNeighborhood::get<1> (
219  int queryId,
220  int nElPerDim,
221  int offset,
222  typename NeighborhoodType<1>::ptr first,
223  typename NeighborhoodType<1>::ptr last)
224 {
225  if ( queryId < 0 || queryId >= nElPerDim ) {
226  std::fill(first, last, -1);
227  } else {
228  *first++ = (queryId > 0) ? queryId-1 : -1;
229  *first++ = queryId;
230  *first++ = (queryId < nElPerDim-1) ? queryId+1 : -1;
231  }
232 }
233 template<>
234 void
235 OneRingNeighborhood::get<2> (
236  int queryId,
237  int nElPerDim,
238  int offset,
239  typename NeighborhoodType<2>::ptr first,
240  typename NeighborhoodType<2>::ptr last)
241 {
242  int offQueryId = queryId - offset;
243  if ( offset < 0 || offset >= nElPerDim*nElPerDim*nElPerDim || offQueryId < 0 || offQueryId >= nElPerDim*nElPerDim ) {
244  std::fill(first, last, -1);
245  } else {
246  std::div_t d = std::div( offQueryId, nElPerDim );
247 
248  // previous row
249  if ( d.quot == 0 ) {
250  std::fill(first, first + 3, -1); first+=3;
251  }
252  else {
253  *first++ = (d.rem > 0) ? queryId-1-nElPerDim : -1;
254  *first++ = queryId - nElPerDim;
255  *first++ = (d.rem < nElPerDim-1) ? queryId+1-nElPerDim : -1;
256  }
257 
258  // current row
259  *first++ = (d.rem > 0) ? queryId-1: -1;
260  *first++ = queryId;
261  *first++ = (d.rem < nElPerDim-1) ? queryId+1 : -1;
262 
263  if ( d.quot+1 >= nElPerDim ) {
264  std::fill(first, first + 3, -1); first+=3;
265  }
266  else {
267  *first++ = (d.rem > 0) ? queryId-1+nElPerDim : -1;
268  *first++ = queryId + nElPerDim;
269  *first++ = (d.rem < nElPerDim-1) ? queryId+1+nElPerDim : -1;
270  }
271  }
272 }
273 template<>
274 void
275 OneRingNeighborhood::get<3> (
276  int queryId,
277  int nElPerDim,
278  int /*offset*/,
279  typename NeighborhoodType<3>::ptr first,
280  typename NeighborhoodType<3>::ptr /*last*/)
281 {
282  int sliceSize = nElPerDim*nElPerDim;
283  int neiSliceSize = 9;
284  std::div_t d = std::div( queryId, sliceSize );
285  get<2>( queryId - sliceSize, nElPerDim, sliceSize*(d.quot - 1), first , first + neiSliceSize );
286  get<2>( queryId , nElPerDim, sliceSize*(d.quot ), first + neiSliceSize, first + 2*neiSliceSize );
287  get<2>( queryId + sliceSize, nElPerDim, sliceSize*(d.quot + 1), first + 2*neiSliceSize, first + 3*neiSliceSize );
288 }
