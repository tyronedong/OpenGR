47  /// Finds congruent candidates in the set Q, given the invariants and threshold distances.
48  /// Returns true if a non empty set can be found, false otherwise.
49  /// @param invariant1 [in] The first invariant corresponding to the set P_pairs
50  /// of pairs, previously extracted from Q.
51  /// @param invariant2 [in] The second invariant corresponding to the set
52  /// Q_pairs of pairs, previously extracted from Q.
53  /// @param [in] distance_threshold1 The distance for verification.
54  /// @param [in] distance_threshold2 The distance for matching middle points due
55  /// to the invariants (See the paper for e1, e2).
56  /// @param [in] First_pairs The first set of pairs found in Q.
57  /// @param [in] Second_pairs The second set of pairs found in Q.
58  /// @param [out] quadrilaterals The set of congruent quadrilateral. In fact,
59  /// it's a super set from which we extract the real congruent set.
60  inline bool FindCongruentQuadrilaterals(
61  Scalar invariant1,
62  Scalar invariant2,
63  Scalar /*distance_threshold1*/,
64  Scalar distance_threshold2,
65  const std::vector <std::pair<int, int>> &First_pairs,
66  const std::vector <std::pair<int, int>> &Second_pairs,
67  Traits4pcs::Set* quadrilaterals) const {
68  using VectorType = gr::Point3D::VectorType;
69 
70  if (quadrilaterals == nullptr) return false;
71 
72  size_t number_of_points = 2 * First_pairs.size();
73 
74  // We need a temporary kdtree to store the new points corresponding to
75  // invariants in the First_pairs and then query them (for range search) for all
76  // the new points corresponding to the invariants in Second_pairs.
77  quadrilaterals->clear();
78 
79  std::vector<VectorType> invariant1Set;
80  invariant1Set.reserve(number_of_points);
81 
82  // build invariants for the first pair set
83  for (size_t i = 0; i < First_pairs.size(); ++i) {
84  const VectorType &p1 = mySampled_Q_3D_[First_pairs[i].first].pos();
85  const VectorType &p2 = mySampled_Q_3D_[First_pairs[i].second].pos();
86  invariant1Set.push_back(p1 + invariant1 * (p2 - p1));
87  }
88 
89  VectorType query;
90  for (size_t i = 0; i < Second_pairs.size(); ++i) {
91  const VectorType &p1 = mySampled_Q_3D_[Second_pairs[i].first].pos();
92  const VectorType &p2 = mySampled_Q_3D_[Second_pairs[i].second].pos();
93 
94  query = p1 + invariant2 * (p2 - p1);
95  for (size_t j = 0; j < invariant1Set.size(); ++j) {
96  const auto&other = invariant1Set[j];
97  if ( (query - other).squaredNorm() < distance_threshold2 )
98  quadrilaterals->push_back(
99  { First_pairs[j].first,
100  First_pairs[j].second,
101  Second_pairs[i].first,
102  Second_pairs[i].second });
103  }
104  }
105 
106  return quadrilaterals->size() != 0;
107  }

109  /// Constructs pairs of points in Q, corresponding to a single pair in the
110  /// in basein P.
111  /// @param [in] pair_distance The distance between the pairs in P that we have
112  /// to match in the pairs we select from Q.
113  /// @param [in] pair_normal_distance The angle between the normals of the pair
114  /// in P.
115  /// @param [in] pair_distance_epsilon Tolerance on the pair distance. We allow
116  /// candidate pair in Q to have distance of
117  /// pair_distance+-pair_distance_epsilon.
118  /// @param [in] base_point1 The index of the first point in P.
119  /// @param [in] base_point2 The index of the second point in P.
120  /// @param [out] pairs A set of pairs in Q that match the pair in P with
121  /// respect to distance and normals, up to the given tolerance.
122  inline void ExtractPairs(Scalar pair_distance,
123  Scalar pair_normals_angle,
124  Scalar pair_distance_epsilon,
125  int base_point1,
126  int base_point2,
127  PairsVector* pairs) const {
128  if (pairs == nullptr) return;
129 
130  pairs->clear();
131  pairs->reserve(2 * mySampled_Q_3D_.size());
132 
133  PairFilterFunctor fun;
134 
135  // Go over all ordered pairs in Q.
136  for (size_t j = 0; j < mySampled_Q_3D_.size(); ++j) {
137  const Point3D& p = mySampled_Q_3D_[j];
138  for (size_t i = j + 1; i < mySampled_Q_3D_.size(); ++i) {
139  const Point3D& q = mySampled_Q_3D_[i];
140 #ifndef MULTISCALE
141  // Compute the distance and two normal angles to ensure working with
142  // wrong orientation. We want to verify that the angle between the
143  // normals is close to the angle between normals in the base. This can be
144  // checked independent of the full rotation angles which are not yet
145  // defined by segment matching alone..
146  const Scalar distance = (q.pos() - p.pos()).norm();
147  if (std::abs(distance - pair_distance) > pair_distance_epsilon) continue;
148 #endif
149 
150  std::pair<bool,bool> res = fun(p,q, pair_normals_angle, myBase_3D_[base_point1],myBase_3D_[base_point2], myOptions_);
151  if (res.first)
152  pairs->emplace_back(i, j);
153  if (res.second)
154  pairs->emplace_back(j, i);
155  }
156  }
157  }
