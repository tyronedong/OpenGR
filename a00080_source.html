117  /// Finds congruent candidates in the set Q, given the invariants and threshold
118  /// distances. Returns true if a non empty set can be found, false otherwise.
119  /// @param invariant1 [in] The first invariant corresponding to the set P_pairs
120  /// of pairs, previously extracted from Q.
121  /// @param invariant2 [in] The second invariant corresponding to the set
122  /// Q_pairs of pairs, previously extracted from Q.
123  /// @param [in] distance_threshold1 The distance for verification.
124  /// @param [in] distance_threshold2 The distance for matching middle points due
125  /// to the invariants (See the paper for e1, e2).
126  /// @param [in] First_pairs The first set of pairs found in Q.
127  /// @param [in] Second_pairs The second set of pairs found in Q.
128  /// @param [out] quadrilaterals The set of congruent quadrilateral. In fact,
129  /// it's a super set from which we extract the real congruent set.
130  inline bool FindCongruentQuadrilaterals(
131  Scalar invariant1,
132  Scalar invariant2,
133  Scalar /*distance_threshold1*/,
134  Scalar distance_threshold2,
135  const std::vector<std::pair<int, int>>& First_pairs,
136  const std::vector<std::pair<int, int>>& Second_pairs,
137  Traits4pcs::Set* quadrilaterals) const {
138 
139  typedef typename PairCreationFunctorType::Point Point;
140 
141 #ifdef SUPER4PCS_USE_CHEALPIX
142  typedef gr::IndexedNormalHealSet IndexedNormalSet3D;
143 #else
144  typedef gr::IndexedNormalSet
145  < Point, //! \brief Point type used internally
146  3, //! \brief Nb dimension
147  7, //! \brief Nb cells/dim normal
148  Scalar> //! \brief Scalar type
149  IndexedNormalSet3D;
150 #endif
151 
152 
153  if (quadrilaterals == nullptr) return false;
154 
155  quadrilaterals->clear();
156 
157  // Compute the angle formed by the two vectors of the basis
158  const Scalar alpha =
159  (myBase_3D_[1].pos() - myBase_3D_[0].pos()).normalized().dot(
160  (myBase_3D_[3].pos() - myBase_3D_[2].pos()).normalized());
161 
162  // 1. Datastructure construction
163  const Scalar eps = pcfunctor_.getNormalizedEpsilon(distance_threshold2);
164 
165  IndexedNormalSet3D nset (eps);
166 
167  for (size_t i = 0; i < First_pairs.size(); ++i) {
168  const Point& p1 = pcfunctor_.points[First_pairs[i].first];
169  const Point& p2 = pcfunctor_.points[First_pairs[i].second];
170  const Point n = (p2 - p1).normalized();
171 
172  nset.addElement((p1+ typename Point::Scalar(invariant1) * (p2 - p1)).eval(), n, i);
173  }
174 
175 
176  std::set< std::pair<unsigned int, unsigned int > > comb;
177 
178  unsigned int j = 0;
179  std::vector<unsigned int> nei;
180  // 2. Query time
181  for (unsigned int i = 0; i < Second_pairs.size(); ++i) {
182  const Point& p1 = pcfunctor_.points[Second_pairs[i].first];
183  const Point& p2 = pcfunctor_.points[Second_pairs[i].second];
184 
185  const VectorType& pq1 = mySampled_Q_3D_[Second_pairs[i].first].pos();
186  const VectorType& pq2 = mySampled_Q_3D_[Second_pairs[i].second].pos();
187 
188  nei.clear();
189 
190  const Point query = p1 + invariant2 * ( p2 - p1 );
191  const VectorType queryQ = pq1 + invariant2 * (pq2 - pq1);
192 
193  const Point queryn = (p2 - p1).normalized();
194 
195  nset.getNeighbors( query, queryn, alpha, nei);
196 
197 
198  VectorType invPoint;
199  //const Scalar distance_threshold2s = distance_threshold2 * distance_threshold2;
200  for (unsigned int k = 0; k != nei.size(); k++){
201  const int id = nei[k];
202 
203  const VectorType& pp1 = mySampled_Q_3D_[First_pairs[id].first].pos();
204  const VectorType& pp2 = mySampled_Q_3D_[First_pairs[id].second].pos();
205 
206  invPoint = pp1 + (pp2 - pp1) * invariant1;
207 
208  // use also distance_threshold2 for inv 1 and 2 in 4PCS
209  if ((queryQ-invPoint).squaredNorm() <= distance_threshold2){
210  comb.emplace(id, i);
211  }
212  }
213  }
214 
215  for (std::set< std::pair<unsigned int, unsigned int>>::const_iterator it = comb.cbegin();
216  it != comb.cend(); it++) {
217  const unsigned int & id = (*it).first;
218  const unsigned int & i = (*it).second;
219 
220  quadrilaterals->push_back( {First_pairs[id].first, First_pairs[id].second,
221  Second_pairs[i].first, Second_pairs[i].second });
222  }
223 
224  return quadrilaterals->size() != 0;
225  }
226 
227  };
228 }
