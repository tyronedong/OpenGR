45 typename MATCH_BASE_TYPE::Scalar
46 MATCH_BASE_TYPE::MeanDistance() const {
47  const Scalar kDiameterFraction = 0.2;
48  using RangeQuery = gr::KdTree<Scalar>::RangeQuery<>;
49 
50  int number_of_samples = 0;
51  Scalar distance = 0.0;
52 
53  for (size_t i = 0; i < sampled_P_3D_.size(); ++i) {
54 
55  RangeQuery query;
56  query.sqdist = P_diameter_ * kDiameterFraction;
57  query.queryPoint = sampled_P_3D_[i].pos().template cast<Scalar>();
58 
59  auto resId = kd_tree_.doQueryRestrictedClosestIndex(query , i).first;
60 
61  if (resId != gr::KdTree<Scalar>::invalidIndex()) {
62  distance += (sampled_P_3D_[i].pos() - sampled_P_3D_[resId].pos()).norm();
63  number_of_samples++;
64  }
65  }
66 
67  return distance / number_of_samples;
68 }

72 MATCH_BASE_TYPE::SelectRandomTriangle(int &base1, int &base2, int &base3) {
73  int number_of_points = sampled_P_3D_.size();
74  base1 = base2 = base3 = -1;
75 
76  // Pick the first point at random.
77  int first_point = randomGenerator_() % number_of_points;
78 
79  const Scalar sq_max_base_diameter_ = max_base_diameter_*max_base_diameter_;
80 
81  // Try fixed number of times retaining the best other two.
82  Scalar best_wide = 0.0;
83  for (int i = 0; i < kNumberOfDiameterTrials; ++i) {
84  // Pick and compute
85  const int second_point = randomGenerator_() % number_of_points;
86  const int third_point = randomGenerator_() % number_of_points;
87  const VectorType u =
88  sampled_P_3D_[second_point].pos() -
89  sampled_P_3D_[first_point].pos();
90  const VectorType w =
91  sampled_P_3D_[third_point].pos() -
92  sampled_P_3D_[first_point].pos();
93  // We try to have wide triangles but still not too large.
94  Scalar how_wide = (u.cross(w)).norm();
95  if (how_wide > best_wide &&
96  u.squaredNorm() < sq_max_base_diameter_ &&
97  w.squaredNorm() < sq_max_base_diameter_) {
98  best_wide = how_wide;
99  base1 = first_point;
100  base2 = second_point;
101  base3 = third_point;
102  }
103  }
104  return base1 != -1 && base2 != -1 && base3 != -1;
105 }

124 bool
125 MATCH_BASE_TYPE::ComputeRigidTransformation(const Coordinates& ref,
126  const Coordinates& candidate,
127  const Eigen::Matrix<Scalar, 3, 1>& centroid1,
128  Eigen::Matrix<Scalar, 3, 1> centroid2,
129  Eigen::Ref<MatrixType> transform,
130  Scalar& rms_,
131  bool computeScale ) const {
132  static const Scalar pi = std::acos(-1);
133 
134  rms_ = std::numeric_limits<Scalar>::max();
135 
136  Scalar kSmallNumber = 1e-6;
137 
138  // We only use the first 3 pairs. This simplifies the process considerably
139  // because it is the planar case.
140 
141  const VectorType& p0 = ref[0].pos();
142  const VectorType& p1 = ref[1].pos();
143  const VectorType& p2 = ref[2].pos();
144  VectorType q0 = candidate[0].pos();
145  VectorType q1 = candidate[1].pos();
146  VectorType q2 = candidate[2].pos();
147 
148  Scalar scaleEst (1.);
149 
150  // Compute scale factor if needed
151  if (computeScale){
152  const VectorType& p3 = ref[3].pos();
153  const VectorType& q3 = candidate[3].pos();
154 
155  const Scalar ratio1 = (p1 - p0).norm() / (q1 - q0).norm();
156  const Scalar ratio2 = (p3 - p2).norm() / (q3 - q2).norm();
157 
158  const Scalar ratioDev = std::abs(ratio1/ratio2 - Scalar(1.)); // deviation between the two
159  const Scalar ratioMean = (ratio1+ratio2)/Scalar(2.); // mean of the two
160 
161  if ( ratioDev > Scalar(0.1) )
162  return std::numeric_limits<Scalar>::max();
163 
164 
165  //Log<LogLevel::Verbose>( ratio1, " ", ratio2, " ", ratioDev, " ", ratioMean);
166  scaleEst = ratioMean;
167 
168  // apply scale factor to q
169  q0 = q0*scaleEst;
170  q1 = q1*scaleEst;
171  q2 = q2*scaleEst;
172  centroid2 *= scaleEst;
173  }
174 
175  VectorType vector_p1 = p1 - p0;
176  if (vector_p1.squaredNorm() == 0) return std::numeric_limits<Scalar>::max();
177  vector_p1.normalize();
178  VectorType vector_p2 = (p2 - p0) - ((p2 - p0).dot(vector_p1)) * vector_p1;
179  if (vector_p2.squaredNorm() == 0) return std::numeric_limits<Scalar>::max();
180  vector_p2.normalize();
181  VectorType vector_p3 = vector_p1.cross(vector_p2);
182 
183  VectorType vector_q1 = q1 - q0;
184  if (vector_q1.squaredNorm() == 0) return std::numeric_limits<Scalar>::max();
185  vector_q1.normalize();
186  VectorType vector_q2 = (q2 - q0) - ((q2 - q0).dot(vector_q1)) * vector_q1;
187  if (vector_q2.squaredNorm() == 0) return std::numeric_limits<Scalar>::max();
188  vector_q2.normalize();
189  VectorType vector_q3 = vector_q1.cross(vector_q2);
190 
191  //cv::Mat rotation = cv::Mat::eye(3, 3, CV_64F);
192  Eigen::Matrix<Scalar, 3, 3> rotation = Eigen::Matrix<Scalar, 3, 3>::Identity();
193 
194  Eigen::Matrix<Scalar, 3, 3> rotate_p;
195  rotate_p.row(0) = vector_p1;
196  rotate_p.row(1) = vector_p2;
197  rotate_p.row(2) = vector_p3;
198 
199  Eigen::Matrix<Scalar, 3, 3> rotate_q;
200  rotate_q.row(0) = vector_q1;
201  rotate_q.row(1) = vector_q2;
202  rotate_q.row(2) = vector_q3;
203 
204  rotation = rotate_p.transpose() * rotate_q;
205 
206 
207  // Discard singular solutions. The rotation should be orthogonal.
208  if (((rotation * rotation).diagonal().array() - Scalar(1) > kSmallNumber).any())
209  return false;
210 
211  //Filter transformations.
212  // \fixme Need to consider max_translation_distance and max_scale too
213  if (options_.max_angle >= 0) {
214  Scalar mangle = options_.max_angle * pi / 180.0;
215  // Discard too large solutions (todo: lazy evaluation during boolean computation
216  if (! (
217  std::abs(std::atan2(rotation(2, 1), rotation(2, 2)))
218  <= mangle &&
219 
220  std::abs(std::atan2(-rotation(2, 0),
221  std::sqrt(std::pow(rotation(2, 1),2) +
222  std::pow(rotation(2, 2),2))))
223  <= mangle &&
224 
225  std::abs(atan2(rotation(1, 0), rotation(0, 0)))
226  <= mangle
227  ))
228  return false;
229  }
230 
231 
232  //FIXME
233  // Compute rms and return it.
234  rms_ = Scalar(0.0);
235  {
236  VectorType first, transformed;
237 
238  //cv::Mat first(3, 1, CV_64F), transformed;
239  for (int i = 0; i < 3; ++i) {
240  first = scaleEst*candidate[i].pos() - centroid2;
241  transformed = rotation * first;
242  rms_ += (transformed - ref[i].pos() + centroid1).norm();
243  }
244  }
245 
246  rms_ /= Scalar(ref.size());
247 
248  Eigen::Transform<Scalar, 3, Eigen::Affine> etrans (Eigen::Transform<Scalar, 3, Eigen::Affine>::Identity());
249  transform = etrans
250  .scale(scaleEst)
251  .translate(centroid1)
252  .rotate(rotation)
253  .translate(-centroid2)
254  .matrix();
255 
256  return true;
257 }

262 void MATCH_BASE_TYPE::init(const std::vector<Point3D>& P,
263  const std::vector<Point3D>& Q,
264  const Sampler& sampler){
265 
266  centroid_P_ = VectorType::Zero();
267  centroid_Q_ = VectorType::Zero();
268 
269  sampled_P_3D_.clear();
270  sampled_Q_3D_.clear();
271 
272  // prepare P
273  if (P.size() > options_.sample_size){
274  sampler(P, options_, sampled_P_3D_);
275  }
276  else
277  {
278  Log<LogLevel::ErrorReport>( "(P) More samples requested than available: use whole cloud" );
279  sampled_P_3D_ = P;
280  }
281 
282 
283 
284  // prepare Q
285  if (Q.size() > options_.sample_size){
286  std::vector<Point3D> uniform_Q;
287  sampler(Q, options_, uniform_Q);
288 
289 
290  std::shuffle(uniform_Q.begin(), uniform_Q.end(), randomGenerator_);
291  size_t nbSamples = std::min(uniform_Q.size(), options_.sample_size);
292  auto endit = uniform_Q.begin(); std::advance(endit, nbSamples );
293  std::copy(uniform_Q.begin(), endit, std::back_inserter(sampled_Q_3D_));
294  }
295  else
296  {
297  Log<LogLevel::ErrorReport>( "(Q) More samples requested than available: use whole cloud" );
298  sampled_Q_3D_ = Q;
299  }
300 
301 
302  // center points around centroids
303  auto centerPoints = [](std::vector<Point3D>&container,
304  VectorType& centroid){
305  for(const auto& p : container) centroid += p.pos();
306  centroid /= Scalar(container.size());
307  for(auto& p : container) p.pos() -= centroid;
308  };
309  centerPoints(sampled_P_3D_, centroid_P_);
310  centerPoints(sampled_Q_3D_, centroid_Q_);
311 
312 
313  initKdTree();
314  // Compute the diameter of P approximately (randomly). This is far from being
315  // Guaranteed close to the diameter but gives good results for most common
316  // objects if they are densely sampled.
317  P_diameter_ = 0.0;
318  for (int i = 0; i < kNumberOfDiameterTrials; ++i) {
319  int at = randomGenerator_() % sampled_Q_3D_.size();
320  int bt = randomGenerator_() % sampled_Q_3D_.size();
321 
322  Scalar l = (sampled_Q_3D_[bt].pos() - sampled_Q_3D_[at].pos()).norm();
323  if (l > P_diameter_) {
324  P_diameter_ = l;
325  }
326  }
327 
328  // Mean distance and a bit more... We increase the estimation to allow for
329  // noise, wrong estimation and non-uniform sampling.
330  P_mean_distance_ = MeanDistance();
331 
332  // Normalize the delta (See the paper) and the maximum base distance.
333  // delta = P_mean_distance_ * delta;
334  max_base_diameter_ = P_diameter_; // * estimated_overlap_;
335 
336  transform_ = Eigen::Matrix<Scalar, 4, 4>::Identity();
337 
338  // call Virtual handler
339  Initialize(P,Q);
340 }
341 
342 }
