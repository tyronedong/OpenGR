152 
153  // Compute the distance and two normal angles to ensure working with
154  // wrong orientation. We want to verify that the angle between the
155  // normals is close to the angle between normals in the base. This can be
156  // checked independent of the full rotation angles which are not yet
157  // defined by segment matching alone..
158  const Scalar distance = (q.pos() - p.pos()).norm();
159 #ifndef MULTISCALE
160  if (std::abs(distance - pair_distance) > pair_distance_epsilon) return;
161 #endif
162  FilterFunctor fun;
163  std::pair<bool,bool> res = fun(p,q, pair_normals_angle, base_3D_[base_point1_],base_3D_[base_point2_], options_);
164  if (res.first)
165  pairs->emplace_back(i, j);
166  if (res.second)
